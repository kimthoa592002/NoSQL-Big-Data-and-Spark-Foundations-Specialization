### 1. Exercise 1: Initialize a Standalone Spark Cluster
#### 1.1 Download Sample Data
   wget https://cf-courses-data.s3.us.cloud-object-storage.appdomain.cloud/IBM-BD0225EN-SkillsNetwork/labs/data/cars.csv

#### 1.2 Initialize the Cluster
***+ Stop any previously running containers***

for i in `docker ps | awk '{print $1}' | grep -v CONTAINER`; do docker kill $i; done

***+ Remove old containers (ignore "No such container" errors)***

docker rm spark-master spark-worker-1 spark-worker-2

***+ Start the Spark Master***

docker run --name spark-master -h spark-master -e ENABLE_INIT_DAEMON=false -p 4040:4040 -p 8080:8080 -v `pwd`:/home/root -d bde2020/spark-master:3.1.1-hadoop3.2

***+ Start Spark Worker 1***

docker run --name spark-worker-1 --link spark-master:spark-master -e ENABLE_INIT_DAEMON=false -p 8081:8081 -v `pwd`:/home/root -d bde2020/spark-worker:3.1.1-hadoop3.2

#### 1.3 Connect PySpark Shell
***+ Launch PySpark connected to the Master***

docker exec -it spark-master /spark/bin/pyspark --master spark://spark-master:7077

***+ Create a DataFrame and cache it***

df = spark.read.csv("/home/root/cars.csv", header=True, inferSchema=True).repartition(32).cache()
df.show()

### 2. Exercise 2: SQL Queries and Debugging via Spark UI
#### 2.1 Run a Faulty Query
from pyspark.sql.functions import udf
import time

@udf("string")
def engine(cylinders):
    time.sleep(0.2) 
    eng = {6: "V6", 8: "V8"} 
    return eng[cylinders]

df = df.withColumn("engine", engine("cylinders"))
dfg = df.groupby("cylinders")
dfa = dfg.agg({"mpg": "avg", "engine": "first"})
dfa.show()

#### 2.2 Debugging

#### 2.3 Fix the Bug
@udf("string")
def engine(cylinders):
    time.sleep(0.2)
    eng = {4: "inline-four", 6: "V6", 8: "V8"}
    return eng.get(cylinders, "other")

***+ Re-run query***
df = df.withColumn("engine", engine("cylinders"))
dfg = df.groupby("cylinders")
dfa = dfg.agg({"mpg": "avg", "engine": "first"})
dfa.show()
